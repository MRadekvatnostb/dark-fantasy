public class utils.ListItem {
    public utils.GameItem getRandomHeal() { // получение рандомной хилки
        int choice = go.Main.rand.nextInt(3);
        return switch (choice) {
            case 0 -> new utils.GameItem.Builder("Склянка йода")
                    .build();
            case 1 -> new utils.GameItem.Builder("Аптечный бинт")
                    .build();
            default -> new utils.GameItem.Builder("Святая вода")
                    .build();
        };
    }
    public utils.GameItem getRandomFood() { // получение рандомной еды
        int choice = go.Main.rand.nextInt(3);
        return switch (choice) {
            case 0 -> new utils.GameItem.Builder("Плесневелый хлеб")
                    .build();
            case 1 -> new utils.GameItem.Builder("Оливье")
                    .build();
            default -> new utils.GameItem.Builder("Сух-паек")
                    .build();
        };
    }
    public utils.GameItem getRandomArtefact() { // получение рандомного артефакта
        int choice = go.Main.rand.nextInt(11);
        return switch (choice) {
            case 0 -> new utils.GameItem.Builder("Ржавый ключ")
                    .setDmgBuff(15)
                    .setHealBuff(15)
                    .build();
            case 1 -> new utils.GameItem.Builder("Массивная труба")
                    .setDmgBuff(10)
                    .build();
            case 2 -> new utils.GameItem.Builder("Провод электрика")
                    .setStanBuff(15)
                    .setCritBuff(5)
                    .setCritChanceBuff(5)
                    .build();
            case 3 -> new utils.GameItem.Builder("Мантия безликого")
                    .setIgnoreDmgBuff(15)
                    .setDmgBuff(7)
                    .build();
            case 4 -> new utils.GameItem.Builder("Копыто конеголового")
                    .setMoneyBuff(20)
                    .build();
            case 5 -> new utils.GameItem.Builder("???")
                    .setCritBuff(20)
                    .build();
            case 6 -> new utils.GameItem.Builder("Мешок собирателя душ")
                    .setStillHpBuff(10)
                    .build();
            case 7 -> new utils.GameItem.Builder("Глаз филина")
                    .setCritChanceBuff(15)
                    .build();
            case 8 -> new utils.GameItem.Builder("Ветвь лешего")
                    .setIgnoreDmgBuff(15)
                    .setHealBuff(20)
                    .build();
            case 9 -> new utils.GameItem.Builder("Зубы пираньи")
                    .setDoubleAttackBuff(15)
                    .build();
            default -> new utils.GameItem.Builder("Крыло падшего ангела")
                    .setResurrectionBuff(30)
                    .build();
        };
    }
}
